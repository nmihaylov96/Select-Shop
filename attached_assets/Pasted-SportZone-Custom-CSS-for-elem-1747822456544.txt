            SportZone
            
                Вашият партньор в света на спорта
            
            
                Ние сме водещ онлайн магазин за спортни стоки в България, посветени на предоставянето на качествени продукти и изключително обслужване.
            
        

        
        
            
                
                    
                        
                    
                    Нашата мисия
                    
                        Да вдъхновяваме и подкрепяме активния начин на живот чрез предоставяне на достъпни, качествени спортни стоки за всеки.
                    
                
                
                
                    
                        
                    
                    Нашето виждане
                    
                        Да станем най-предпочитаната дестинация за спортни ентусиасти в региона, предлагайки иновативни продукти и персонализирани решения.
                    
                
                
                
                    
                        
                    
                    Нашите ценности
                    
                        Страст към спорта, професионализъм, честност и ангажимент към клиентите са в основата на всичко, което правим.
                    
                
            
        

        
        
            Нашата история
            
            
                
                
                    
                    
                        
                        
                            2010 - Основаване
                            
                                SportZone е основан от група ентусиазирани спортисти с цел да предостави достъп до качествени спортни стоки на конкурентни цени.
                            
                        
                    
                    
                    
                    
                        
                        
                            2012 - Първи физически магазин
                            
                                Отваряме врати на първия ни физически магазин в София, което ни позволява да достигнем до повече клиенти.
                            
                        
                    
                    
                    
                    
                        
                        
                            2015 - Онлайн платформа
                            
                                Стартираме нашата онлайн платформа, която бързо се превръща в основен източник на приходи за компанията.
                            
                        
                    
                    
                    
                    
                        
                        
                            2018 - Разширяване
                            
                                Разширяваме дейността си с още 3 магазина в големи градове и подобряваме логистичната ни мрежа.
                            
                        
                    
                    
                    
                    
                        
                        
                            2023 - Лидер в сектора
                            
                                Днес сме лидер в онлайн продажбите на спортни стоки в България с над 50,000 доволни клиенти.

        // Simple fade-in animation when page loads
        document.addEventListener('DOMContentLoaded', function() {
            document.body.classList.add('fade-in');
            
            // Category filter functionality
            const categoryBtns = document.querySelectorAll('.category-btn');
            const productCards = document.querySelectorAll('.product-card');
            
            categoryBtns.forEach(btn => {
                btn.addEventListener('click', function() {
                    // Remove active class from all buttons
                    categoryBtns.forEach(b => b.classList.remove('active'));
                    
                    // Add active class to clicked button
                    this.classList.add('active');
                    
                    const category = this.dataset.category;
                    
                    // Show/hide products based on category
                    productCards.forEach(card => {
                        if (category === 'all' || card.dataset.category === category) {
                            card.style.display = 'block';
                        } else {
                            card.style.display = 'none';
                        }
                    });
                });
            });
            
            // Simple search functionality
            const searchInput = document.getElementById('search');
            searchInput.addEventListener('keyup', function() {
                const searchTerm = this.value.toLowerCase();
                
                productCards.forEach(card => {
                    const title = card.querySelector('h3').textContent.toLowerCase();
                    const description = card.querySelector('p').textContent.toLowerCase();
                    
                    if (title.includes(searchTerm) || description.includes(searchTerm)) {
                        card.style.display = 'block';
                    } else {
                        card.style.display = 'none';
                    }
                });
            });
        });

                        Как мога да направя поръчка?
                        
                    
                    
                        
                            Можете да направите поръчка директно от нашия уебсайт, като изберете желаните продукти и следвате стъпките в кошницата. Също така можете да направите поръчка по телефона на +359 2 123 4567 в работното време.
                        
                    
                

                
                
                    
                        Какви са методите за плащане?
                        
                    
                    
                        
                            Приемаме следните методи на плащане: кредитни/дебитни карти (Visa, Mastercard, American Express), банков превод и наложен платеж. Всички онлайн плащания се обработват сигурно през нашата платежна система.
                        
                    
                

                
                
                    
                        Какво е времето за доставка?
                        
                    
                    
                        
                            За София доставките обикновено се извършват в рамките на 1 работен ден. За останалата част на страната времето за доставка е 2-3 работни дни. При поръчки направени в петък следобед или през уикенда, доставките се извършват в понеделник.

        // Simple fade-in animation when page loads
        document.addEventListener('DOMContentLoaded', function() {
            document.body.classList.add('fade-in');
            
            // FAQ toggle functionality
            const faqToggles = document.querySelectorAll('.faq-toggle');
            
            faqToggles.forEach(toggle => {
                toggle.addEventListener('click', function() {
                    const content = this.nextElementSibling;
                    const icon = this.querySelector('i');
                    
                    // Toggle content visibility
                    content.classList.toggle('hidden');
                    
                    // Rotate icon
                    if (content.classList.contains('hidden')) {
                        icon.style.transform = 'rotate(0deg)';
                    } else {
                        icon.style.transform = 'rotate(180deg)';
                    }
                    
                    // Close other open FAQs
                    faqToggles.forEach(otherToggle => {
                        if (otherToggle !== this) {
                            otherToggle.nextElementSibling.classList.add('hidden');
                            otherToggle.querySelector('i').style.transform = 'rotate(0deg)';
                        }
                    });
                });
            });
            
            // Form validation
            const form = document.querySelector('form');
            if (form) {
                form.addEventListener('submit', function(e) {
                    const name = document.getElementById('name');
                    const email = document.getElementById('email');
                    const agree = document.getElementById('agree');
                    
                    if (!name.value || !email.value || !agree.checked) {
                        e.preventDefault();
                        alert('Моля, попълнете всички задължителни полета и съгласете се с условията!');
                    }
                });
            }
        });
